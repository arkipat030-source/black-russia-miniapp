function submitForm() { const nickname = document.getElementById('nickname').value; const serverElement = document.getElementById('server'); const serverText = serverElement.options[serverElement.selectedIndex].text; if(!nickname) { alert("Пожалуйста, введите ваш никнейм!"); return; } // Собираем данные в объект const dataToSend = { nickname: nickname, server: serverText }; // ОТПРАВКА ДАННЫХ В БОТА (Важная строка!) tg.sendData(JSON.stringify(dataToSend)); // Переключение на экран с пульсирующей зеленой галочкой const card = document.getElementById('recoveryCard'); const success = document.getElementById('successScreen'); card.style.opacity = '0'; setTimeout(() => { card.style.display = 'none'; success.style.display = 'block'; setTimeout(() => { success.style.opacity = '1'; }, 50); }, 500); }
